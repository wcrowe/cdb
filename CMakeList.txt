cmake_minimum_required(VERSION 3.14)
project(dbview LANGUAGES C)

# Use C11 standard
set(CMAKE_C_STANDARD 11)
set(CMAKE_C_STANDARD_REQUIRED ON)

# Debug symbols + all warnings
add_compile_options(-g -Wall -Wextra -Werror)

# Create the executable
add_executable(dbview
    src/main.c
    src/file.c
    src/parse.c
)

# Include directory
target_include_directories(dbview PRIVATE include)

# Optional: make clean actually delete everything
set_directory_properties(PROPERTIES
    ADDITIONAL_MAKE_CLEAN_FILES "${CMAKE_BINARY_DIR}/mynewdb.db"
)
